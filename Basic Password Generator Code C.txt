#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <string.h>
#include <stdbool.h>


void generate_password(int length, bool use_upper, bool use_lower, bool use_nums, bool use_syms) {
    const char uppercase[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
    const char lowercase[] = "abcdefghijklmnopqrstuvwxyz";
    const char numbers[] = "0123456789";
    const char symbols[] = "!@#$%^&*()_+-=[]{}|;:,.<>?";

    
    char pool[256] = "";

    if (use_upper) strcat(pool, uppercase);
    if (use_lower) strcat(pool, lowercase);
    if (use_nums)  strcat(pool, numbers);
    if (use_syms)  strcat(pool, symbols);

    int pool_size = strlen(pool);

    if (pool_size == 0) {
        printf("\nError: You must select at least one character type to generate a password.\n");
        return;
    }

    printf("\n====================================\n");
    printf("Generated Password: ");
    
   
    for (int i = 0; i < length; i++) {
        int random_index = rand() % pool_size;
        printf("%c", pool[random_index]);
    }
    
    printf("\n====================================\n\n");
}

int main() {
    int length;
    char choice;
    bool use_upper, use_lower, use_nums, use_syms;

    
    srand((unsigned int)time(NULL)); 

    printf("=== Minimalist Password Generator ===\n\n");

    
    printf("Enter desired password length: ");
    if (scanf("%d", &length) != 1 || length <= 0) {
        printf("Invalid length. Please enter a positive number.\n");
        return 1;
    }

    
    while ((getchar()) != '\n');

   
    printf("Include uppercase letters? (y/n): ");
    scanf("%c", &choice);
    use_upper = (choice == 'y' || choice == 'Y');
    while ((getchar()) != '\n');

    printf("Include lowercase letters? (y/n): ");
    scanf("%c", &choice);
    use_lower = (choice == 'y' || choice == 'Y');
    while ((getchar()) != '\n');

    printf("Include numbers? (y/n): ");
    scanf("%c", &choice);
    use_nums = (choice == 'y' || choice == 'Y');
    while ((getchar()) != '\n');

    printf("Include symbols? (y/n): ");
    scanf("%c", &choice);
    use_syms = (choice == 'y' || choice == 'Y');

    
    generate_password(length, use_upper, use_lower, use_nums, use_syms);

    return 0;
}
